Adicionei à gramática regras para definição de constantes e
parâmetros-referência.

Exemplos:

constantes
  PI : real = 3.141592653589793238;
fim-constantes

//Proponho, também, uma regra para inicialização de tipos compostos,
//como vetores/matrizes e estruturas:

estrutura T
  a: literal;
  b: vetor[3] de inteiros;
fim-estrutura

constantes
  v1 : T = [a: "teste", b: [9,8,7]];
fim-constantes

//Constantes também podem ser usadas para parâmetros-referência

função f( ref constante t: inteiro) ...

//E podem ser declarados localmente:

função f()
 v1: inteiro;
 constante v2 : inteiro = 10;
início
//...
fim
