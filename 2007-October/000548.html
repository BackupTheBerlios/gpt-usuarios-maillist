A abordagem que foi definida para o gpt é similar a isso. Ao "criar" uma
matriz o usuário indica (GptAssembly) o tamanho da informação que a matriz
armanzenará, o número de dimensões e o tamanho inicial de cada dimensão.

| Uma matriz[3] de inteiros seria uma área contigua onde cada célula
| amazenaria um dos inteiros.

Isso mesmo.

| Uma matriz[3] de literais seria uma área contigua capaz de armazenar 3
| ponteiros, que referenciariam os respectivos literais. O mesmo
| aconteceria para matrizes de estruturas e matrizes n-dimensionais (que
| seriam apenas arrays que apontam para arrays ....)

Isso mesmo :-)

| De qualquer forma, uma matriz[10] de inteiros ocupa o mesmo espaço de
| uma matriz[10] de literais e de uma matriz[10] de estruturas.

Exato !!! O que a matriz armazena é a mesma informação que uma variável
inteira ou literal armazena. Isso facilita inclusive para comandos do tipo:

meu_inteiro := matriz_inteira[10];

Porque nesse caso a VM simplesmente executa uma variante de memcpy copiando
os dados diretamente da posição 10 para o endereço da variável. Se é um
inteiro, um literal, ou estrutura, não importa... O que importa é o tamanho
do dado.

| A única diferença de matriz[10] de inteiros das demais matrizes é que
| esta pode ter cada célula com o próprio valor, e não com um ponteiro
| para o valor.

As nossas idéias estão sincronizadas hehehe. O que importa é como a VM
originalmente armazena o tipo de dado.

| Toda a manipulação desta memória ficaria a cargo de funções escritas
| no runtime, e chamadas utilizando libcall. Poderiamos definir que o
| formato da área de memória de uma matriz e de uma estrutura tenha um
| cabeçalho contendo informações sobre o dado, como tamanho e tipo, e
| funções do runtime poderiam obter estas informações.

Existe um cabeçalho sim que contém o tamanho da área alocada, o número de
dimensões, o tamanho de cada informação e o tamanho de cada dimensão. Porém
tanto os dados quanto o cabeçalho são manipulados através de opcodes, ou
seja, a VM tem suporte nativo a matrizes.

// Exemplo em Portugol

algoritmo teste;
variáveis
    m: matriz[10] do tipo inteiro;
    i: inteiro;
fim-variáveis
início
    para i de 1 até 10 faça
        m[i] := i * 10;
    fim-para
    imprima( "Valores das coordenadas:" );
    para i de 1 até 10 faça
        imprima( m[i] );
    fim-para
fim



// Convertido para GptAssembly



program exemplo

var m matrix
var i int

proc main
    var t1 int

    // aloca espaço para a matriz m (int tem 4 bytes nesse exemplo)
    m1alloc m, 4, 10 // dado com 4 bytes, 10 elementos iniciais

    // para i de 1 até 10 faça
    iset i, 1
    teste:
    ile t1, i, 10
    ifnot t1 próximo

    // m[i] := i * 10;
    imul t1, i, 10
    m1set m, i, t1 // m1set (matriz de 1 dimensão), m2set (2 dimensões), ...

    // fim-para
    jmp teste
    próximo:

    // imprima( "Valores armazenados:" );
    push_sp
    push "Valores armazenados:"
    push_string
    push_1
    pcall imprima
    pop_sp

    // para i de 1 até 10 faça
    iset i, 1
    teste2:
    ile t1, i, 10
    ifnot t1, próximo2

    // imprima( m[i] );
    push_sp
    m1get t1, m, i // m1get é um memcpy disfarçado :-)
    push t1
    push_int
    push_1
    pcall imprima
    pop_sp

    // fim-para
    jmp teste2
    próximo2:

    // desaloca espaço da matriz c
    mfree c

    exit_0
end-proc

end-program
