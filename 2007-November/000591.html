> Pode mostrar um exemplo com matrizes com mais de uma dimensão?


Eu imaginei algo como:


// Portugol

algoritmo vetor_na_pilha;

variável resultado: matriz[3][4] do tipo inteiro;

início
    resultado[0][0] := 10;
    resultado[1][2] := 20;
    resultado[2][3] := resultado[0][0] + resultado[1][2];
fim



// Exemplo em GptAssembly

program vetor_na_pilha;

var resultado matrix

proc main
    m2alloc resultado, 4, 12 // 4 bytes, 12 elementos

    // resultado[0][0] := 10;
    m2set resultado, 0, 0, 10

    // resultado[1][2] := 20;
    m2set resultado, 1, 2, 20

    // resultado[2][3] := resultado[0][0] + resultado[1][2];
    m2get t1, resultado, 0, 0
    m2get t2, resultado, 1, 2
    isum t3, t1, t2
    m2set resultado, 2, 3, t3

    mfree resultado
    ret

endproc

endprogram
