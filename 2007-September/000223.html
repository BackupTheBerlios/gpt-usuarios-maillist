Agora, explicando como eu gostaria de oferecer a "libc" do G-Portugol:

//Abordagem #1
//Supondo que os bytecodes abstraem os serviços do SO
//oferecendo pcodes para estes.
//Neste caso, deve haver bytecodes para IO, threads,
//processos, e tudo o mais.

função imprima(i : inteiro)
início
  @bytecode-início
    push i
    pcall print_integer
  @bytecode-fim
fim

função imprima(l : literal)
início
  @bytecode-início
    push l
    pcall print_literal
  @bytecode-fim
fim

//Abordagem #2
//Supondo que os bytecodes não oferecem os serviços oferecidos
//pelo SO. Poderiamos utilizar inline assembly para ter acesso direto:

função imprima(texto : literal)
  tam : inteiro;
início
  tam := tamanho_literal(texto);

  @início-asm
    mov edx, [tam]
    mov eax, 4
    mov ebx, 1
    mov ecx, [texto] //texto
    int 80h
  @fim-asm
fim

função imprima(numero : inteiro)
  texto : literal;
início
  texto := inteiro_para_literal(numero);
  imprima(texto);
fim

função imprima(numero : real) ...
função imprima(lógico : real) ...
função inteiro_para_literal(i: inteiro) ...
função literal_para_inteiro(l: literal) ...

função leia(ref i : inteiro)
  texto : literal;
início
  leia(texto);
  i := literal_para_inteiro(texto)
fim

//exemplo rudimentar com limite de buffer
função leia(ref texto : literal)
  buffer : vetor[1024] de caracteres;
início
  @início-asm
    mov eax, 3
    mov ebx, 0
    mov ecx, [buffer]
    mov edx, [1024] //size do buffer
    int 80h

    mov ebx, [buffer]
    mov [ebx-eax-1], byte 0 //adiciona \0
  @fim-asm

  texto := vetor_para_literal(buffer);
fim

função vetor_para_literal(v : vetor[] de caracteres) : literal
  i : inteiro;
  texto : literal;
início
  i := 0;
  enquanto v[i] <> 0 faça
    texto := texto >> v[i]; //concatenação de caracteres
  fim-enquanto

  retorne texto;
fim

A abordagem #2  era o que eu tinha planejado antes da máquina virtual
ser sugerida. Isso é, eu queria implementar algo análogo a libc.

resolveria alguns problemas de forma bastante elegante, mantendo um
pouco da compatibilidade com a literatura.
